func (s *ChannelMembershipRepo) UnenrollChannelMembership(ctx context.Context, accountID string) error {
	params  :=  db.UpdateMembershipEndTimeParams{
		AccountID: accountID,
		EndAt: sql.NullTime{
			Time: time.Now(),
			Valid: true,
		},
	}
	err  :=  s.queries.UpdateMembershipEndTime(ctx, params)
	if  err  !=  nil {
		return  err
	}
	return nil
}
